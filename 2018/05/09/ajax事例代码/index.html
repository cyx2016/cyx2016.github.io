ajax(jquery写法)

<script>
    $(function () {
        //用户查询
        $("#research").click(function () {
            var role = $("#role").val();
            var id = $("#id").val();
            var name = $("#name").val();
            var sex = $("#sex").val();
            $.ajax({
                url: '/user/showByParams',
                data:'{"id": "' +id+ '","name":"'+name+'","sex":"'+sex+'", "role": "'+role+'"}',
                /*async: true,*/
                type: "POST",
                contentType: "application/json",
                dataType: 'json',
                success: function (date) {+
                    //alert(date[0].id);
                    $("#usertable tbody").html("");
                    var html;
                   /* if(result != []) {
                        var date = eval("(" + eval(result) + ")");
                    }*/
                    for(var i=0;i<date.length;i++){
                        html+= "<tr>"+"<td>"+date[i].id+"</td>"+
                            "<td>" +date[i].name+"</td>"+
                             "<td>"+date[i].password+"</td>"+
                            "<td>"+date[i].role+"</td>"+
                            "<td>"+date[i].develope+"</td>"+
                            "<td>"
                        "<a href=\"userMod.html\" class=\"btn btn-primary btn-sm\">"+"修改"+"</a>"
                        "<a href=\"#\" class=\"btn btn-primary btn-sm\">"+"删除"+"</a>"
                        "<a href=\"#\" class=\"btn btn-primary btn-sm\">"+"详细"+"</a>"+"</td>"
                        +"</tr>";
                        //console.log("执行成功"+result[i].name+html);
                    }
                    $("#usertable tbody").html(html);
                },
                error:function(){
                    alert(11);
                }
            });
        });
    });
